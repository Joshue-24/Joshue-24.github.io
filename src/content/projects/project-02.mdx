### Detección de Espacios Libres

Utilizamos técnicas de procesamiento de imágenes y modelos de clasificación para identificar el estado de cada plaza de estacionamiento (ocupada o libre) en tiempo real. Esto se logra analizando transmisiones de video de cámaras estratégicamente ubicadas.

```python
import cv2
import numpy as np

def detect_parking_spaces(frame, parking_spots_coords):
    # parking_spots_coords: lista de tuplas (x, y, w, h) para cada plaza
    status = {}
    for i, (x, y, w, h) in enumerate(parking_spots_coords):
        spot_roi = frame[y:y+h, x:x+w]
        # Convertir a escala de grises y aplicar umbral para simplificar la detección
        gray_roi = cv2.cvtColor(spot_roi, cv2.COLOR_BGR2GRAY)
        _, threshold_roi = cv2.threshold(gray_roi, 50, 255, cv2.THRESH_BINARY)

        # Contar píxeles blancos (representando el espacio libre)
        white_pixels = cv2.countNonZero(threshold_roi)

        # Un umbral simple para determinar si está ocupado o libre
        if white_pixels < (w * h * 0.1): # Si menos del 10% es blanco, asumimos ocupado
            status[f"spot_{i+1}"] = "occupied"
            cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 0, 255), 2) # Rojo para ocupado
        else:
            status[f"spot_{i+1}"] = "free"
            cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 255, 0), 2) # Verde para libre
    return frame, status
```
